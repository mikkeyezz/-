stack = []

# Добавляем элементы в стек (push)
stack.append('Apple')
stack.append('Banana')
stack.append('Cherry')
stack.append('Orange')

print("Стек:", stack)

# Удаляем и возвращаем верхний элемент (pop)
top_element = stack.pop()
print("Удален верхний элемент:", top_element)
print("Стек после pop:", stack)

# Смотрим верхний элемент без удаления (peek)
print("Текущий верхний элемент:", stack[-1])
